#include <stdio.h>
#include<stdint.h>
void alternate_positive(int32_t arr[], uint32_t u32_n){
    int32_t pos[u32_n],neg[u32_n];
    uint32_t u32_i,u32_j=0,u32_k=0,u32_pos=0,u32_neg=0;
    for(u32_i=0;u32_i<u32_n;u32_i++){
        if(arr[u32_i]>=0){
            pos[u32_pos++]=arr[u32_i];
        }
        if(arr[u32_i]<0){
            neg[u32_neg++]=arr[u32_i];
        }
    }
    
    u32_i=0;
    while((u32_i<u32_pos) && (u32_j<u32_neg)){
        arr[u32_k++] = pos[u32_i++];
        arr[u32_k++] = neg[u32_j++];
    }
    
    while(u32_i<u32_pos){
        arr[u32_k++]=pos[u32_i++];
    }
    while(u32_j<u32_neg){
        arr[u32_k++] = neg[u32_j++];
    }
    for(u32_k=0;u32_k<u32_n;u32_k++){
        printf("%d, ",arr[u32_k]);
    }
}
int main()
{
    int32_t arr[] = {9,5,-2,-1,5,0,-5,-3,2};
    uint32_t u32_n =((sizeof(arr)) / (sizeof(arr[0])));
    alternate_positive(arr,u32_n);

}
